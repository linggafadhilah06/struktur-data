def hot_potato(names, num):
  q = Queue()
  for name in names:
    q.enqueue(name)

  while len(q) > 1:
    for _ in range(num):
      # Oper: dequeue depan, enqueue belakang
      q.enqueue(q.dequeue())
    q.dequeue()  # Tersingkir!

  return q.dequeue()  # Pemenang!
