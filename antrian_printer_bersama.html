# Inisialisasi antrian printer
printer_queue = Queue()

# User mengirim dokumen
printer_queue.enqueue("laporan.pdf")
printer_queue.enqueue("tugas.docx")
printer_queue.enqueue("foto.jpg")

# Printer memproses (FIFO)
while not printer_queue.isEmpty():
  doc = printer_queue.dequeue()
  print(f"Mencetak: {doc}")

# Output:
# Mencetak: laporan.pdf
# Mencetak: tugas.docx
